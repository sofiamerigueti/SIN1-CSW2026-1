            import math

def resolver_equacao_segundo_grau(a, b, c):
    if a == 0:
        return "Não é equação de segundo grau (a deve ser diferente de 0)."
    
    delta = b**2 - 4*a*c
    
    if delta < 0:
        return "A equação não possui raízes reais."
    elif delta == 0:
        x = -b / (2*a)
        return f"A equação possui uma raiz real: x = {x}"
    else:
        x1 = (-b + math.sqrt(delta)) / (2*a)
        x2 = (-b - math.sqrt(delta)) / (2*a)
        return f"A equação possui duas raízes reais: x1 = {x1}, x2 = {x2}"

# Exemplo de uso:

print(resultado)
